	\section{Randomization preserves NIP}
	
	\begin{prop}
		Let $T$ be an arbitrary theory (in continuous logic), and let $M$ be an $\aleph_0$-saturated model of $T$.
		
		Then $T$ has NIP if and only if for every formula $\varphi(x,y)$, the family $\{\varphi(x,c)\mid c\in M \}$ does not contain an $\ell^1$-sequence (with respect to the supremum norm).
	\end{prop}
	
	\begin{cor}
		Suppose $T$ is arbitrary, while $M$ is $\aleph_0$-saturated and $\aleph_0$-strongly homogeneous.
		
		Then $T$ has NIP if and only if $(\Aut(M),S(M))$ is a tame dynamical system.
	\end{cor}
